  className,
  ...props

  width?: never;
  height?: never;
}) {
  return (
    <OptimizedImage
      src={src}

      width={400}

      aspectRatio="4/5"
      priority={priority}

      sizes="(max-width: 640px) 100vw, (max-width: 1024px) 50vw, 400px"

      {...props}
    />
  );
}
